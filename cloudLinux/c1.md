# 改变目录
    - cd /root

# 创建文件
    touch a.txt

# 创建目录
    mkdir
    mkdir -p 没上级 自动创建

# 复制
    cp a.txt b.txt
    cp -r a.txt b.txt

# 移动/改名
    mv  aa/a.txt bb/a.txt
    
# 删除
    - rm 删除
    - rm -rf 批量删

# 查看文件
    - cat 查看全部
    - more entry 一行一行查看
    - more space 一页一页看
    - head -10 a.txt 查看前10页
    - tail -10 a.txt 查看后10页
    - grep maven a.txt （查看含有maven的行数）过滤文件
